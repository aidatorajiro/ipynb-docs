def stft(x, win, step):
    inputlen = len(x) # 
    slicelen = len(win) # 
    flames = int(scipy.ceil(float(inputlen - slicelen + step) / step)) # 
    
    new_x = scipy.zeros(slicelen + ((flames - 1) * step), dtype = scipy.float64)
    new_x[:inputlen] = x # 
    
    X = scipy.zeros([flames, slicelen / 2], dtype = scipy.complex64) # ()
    for i in range(flames):
        start = step * i
        F = scipy.fftpack.fft(new_x[start : start + slicelen] * win)
        X[i, :] = F[:len(F) / 2]
    
    return X

if __name__ == "__main__":
    fs, data = scipy.io.wavfile.read("0002.wav")

    fftLen = 2048 # 
    win = scipy.hamming(fftLen) # 
    step = fftLen / 4

    ### STFT
    spectrogram = stft(data, win, step)

    ### Plot
    pylab.figure(figsize=[20,5])
    pylab.plot(data)

    pylab.title("", fontproperties=fp)

    pylab.xlim([0,400])
    pylab.ylim([0,200])
    pylab.imshow(abs(spectrogram).T, aspect = "auto", origin = "lower")
    pylab.title("", fontproperties=fp)
